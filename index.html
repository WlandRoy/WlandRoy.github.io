swift json 编解码

(官方链接)[https://developer.apple.com/documentation/foundation/archives_and_serialization/using_json_with_custom_types]

简单的解码
  • 如下一个json格式字符串
    let json = """
{
    "name" : "Banana",
    "points" : 200,
}
"""
  • 定义结构体
    struct Product:Codable {
    var name:String
    var points:Int
}
  • 解码
    guard let data = json.data(using: .utf8) else { fatalError("字符串转data失败") }
let decoder = try JSONDecoder().decode(Product.self, from: data)
print(decoder) // Product(name: "Banana", points: 200)
有时候json是一个数组,有的元素对应字段没有如下
let json = """
[
    {
        "name": "Banana",
        "points": 200,
        "description": "A banana grown in Ecuador."
    },
    {
        "name": "Orange",
        "points": 100
    }
]
"""
  • 对应结构体
    struct Product:Codable {
    var name:String
    var points:Int
    var description:String?
}
  • 解码
    guard let data = json.data(using: .utf8) else { fatalError("字符串转data失败") }
let decoder = try JSONDecoder().decode([Product].self, from: data)

print(decoder) // [Product(name: "Banana", points: 200, description: Optional("A banana grown in Ecuador.")), Product(name: "Orange", points: 100, description: nil)]
有时json有字段但是为null
let json = """
{
    "name" : null,
    "points" : null,
}
"""

struct Product:Codable {
    var name:String?
    var points:Int?
}

guard let data = json.data(using: .utf8) else { fatalError("字符串转data失败") }
let decoder = try JSONDecoder().decode(Product.self, from: data)
print(decoder) // Product(name: nil, points: nil)
有时json的字段我们不需要或者想换一个字段表示这是就需要转换一下了
let json = """
[
    {
        "product_name": "Bananas",
        "product_cost": 200,
        "description": "A banana grown in Ecuador."
    },
    {
        "product_name": "Oranges",
        "product_cost": 100,
        "description": "A juicy orange."
    }
]
"""
  • 结构体

    每个声明的字段必须加上哪怕不需要转换 CodingKeys也是固定写死的格式

    struct Product:Codable {
    var productName:String
    var points:Int
    var description:String
    
    // 每个声明的字段必须加上哪怕不需要转换
    private enum CodingKeys:String,CodingKey {
        case productName = "product_name"
        case points = "product_cost"
        case description = "description"
    }
}
  • 解码
    guard let data = json.data(using: .utf8) else { fatalError("字符串转data失败") }
let decoder = try JSONDecoder().decode([Product].self, from: data)
print(decoder)

    var name:String
    var description:String

let jsonObject = try JSONSerialization.jsonObject(with: data, options: [])
print(jsonObject)

let jsonData = try JSONSerialization.data(withJSONObject: ["name":"Orange","description":"A banana grown in Ecuador."])

Swift错误处理

发表于 2024-01-01

  • 定义自动售货机错误处理 定义一个枚举 遵循Error协议

    enum VendingMachineError:Error {
    case invalidSelection // 无效的选择
    case insufficientFunds(coinsNeeded:Int) // 金额不足
    case outOfStock // 缺货
}

  • 定义商品模型

    struct Item {
    var price:Int
    var count:Int
}

  • 定义一个自动售货机

    class VendingMachine {
    var inventory = [
        "Candy Bar":Item(price: 12, count: 7),
        "Chips": Item(price: 10, count: 4),
        "Pretzels": Item(price: 7, count: 11)
    ]
    
    var coinsDeposited = 0
    
    func vend(itemNamed:String) throws {
        // 需要的商品名称没有找到则是无效商品
        guard let item = inventory[itemNamed] else {
            throw VendingMachineError.invalidSelection
        }
        // 商品的数量判断
        guard item.count > 0 else {
            throw VendingMachineError.outOfStock
        }
        // 商品还需要多少硬币
        guard item.price < coinsDeposited else {
            throw VendingMachineError.insufficientFunds(coinsNeeded: (item.price - coinsDeposited))
        }

        print("售出")
    }
    
}

  • 如下用法

    var vendingMachine = VendingMachine()
vendingMachine.coinsDeposited = 8
do {
    try vendingMachine.vend(itemNamed: "Chips")
} catch VendingMachineError.invalidSelection {
    print("没有找到对应的商品")
} catch VendingMachineError.insufficientFunds(coinsNeeded: let coinsNeeded) {
    print("需要\(coinsNeeded)硬币")
} catch VendingMachineError.outOfStock {
    print("缺货")
}

  • 自定义错误类型

    // 自动售货机自定义错误信息
extension VendingMachineError:LocalizedError {
    var errorDescription: String? {
        switch self {
        case .invalidSelection:
            return "无效的选择"
        case .insufficientFunds(coinsNeeded: let coinsNeeded):
            return "还缺少\(coinsNeeded)硬币"
        case .outOfStock:
            return "缺货"
        }
    }
}

  • 用法

    do {
    try vendingMachine.vend(itemNamed: "Chips")
} catch {
    print(error.localizedDescription)
}

Swift枚举小记

发表于 2024-01-01

常用的枚举

  • 定义一个方向枚举 
    enum CompassPoint {
    case north
    case south
    case east
    case west
}

var direction = CompassPoint.east
switch direction {
case .east:
    print("东向")
case .south:
    print("南向")
case .north:
    print("北向")
case .west:
    print("西向")
}

枚举遵循 CaseIterable协议可以遍历自己的元素

  • 定义一个饮料枚举 遵循 CaseIterable协议
    enum Beverage:CaseIterable {
    case coffee,tea,juice
}
  • 变量枚举的每个元素 , 打印枚举元素的个数
    for item in Beverage.allCases {
    print(item)
}
print("\(Beverage.allCases.count) beverages available")

原始值

  • 定义枚举时如果没有指定类型那么其值就是定义的枚举类型否则就是指定类型的值

    如上 CompassPoint 的枚举 定义的 direction 打印的是 east

  • 定义枚举类型
    enum PetType:String {
    case dog = "小狗"
    case cat = "小猫"
}
var pet = PetType.dog
print(pet.rawValue) // 小狗

关联值

  • 定义一个电子码的枚举 分为条形码和二维码两种类型 条形码是四个整数组合的元组,条形码则是字符串
    enum Barcode {
    case upc(Int,Int,Int,Int)
    case qrCode(String)
}

// 创建一个条形码的类型
var barCode = Barcode.upc(8, 85909, 51226, 3)
// 二维码
barCode = Barcode.qrCode("ABCDEKFJSHEHUFHEOP")

switch barCode {
case .upc(let num1, let num2, let num3, let num4):
    print("条形码是\(num1) \(num2) \(num3) \(num4)")
case .qrCode(let code):
    print("二维码的内容是: \(code)")
}

Swift类和结构体小记

发表于 2024-01-01

定义如下一个分辨率的结构体

struct Resolution {
    var width = 0
    var height = 0
}
var hd = Resolution(width: 1920,height: 1080)
var cinema = hd
cinema.width = 2048
print(hd.width) // 1920

上面打印没变2048 , 在swift中结构体(Struct)和枚举(enum)都是值类型,当cinema=hd时两个结构体都是引用同一块内存的值,但当cinema赋予新值时,俩结构体才引用不同内存的值
(官方文档说明)[https://docs.swift.org/swift-book/documentation/the-swift-programming-language/classesandstructures/]

类

  • 定义一个类

    class VideoMode {
    var resolution = Resolution()
    var interlaced = false
    var frameRate = 0.0
    var name:String?
}
    var hd = Resolution(width: 1920,height: 1080)

let tenEighty = VideoMode()
tenEighty.resolution = hd
tenEighty.interlaced = true
tenEighty.name = "1080i"
tenEighty.frameRate = 25.0

let alsoTenEighty = tenEighty
alsoTenEighty.frameRate = 30.0
print(tenEighty.frameRate) // 30.0

    通过打印发现tenEighty.frameRate的值变了,swift中类就是引用类型,并不是值复制

  • 通过 === 可以判断两个对象是否是引用同一块地址

    let tenEighty = VideoMode()
tenEighty.resolution = hd
tenEighty.interlaced = true
tenEighty.name = "1080i"
tenEighty.frameRate = 25.0

let alsoTenEighty = tenEighty
alsoTenEighty.frameRate = 30.0
print(tenEighty.frameRate) // 30.0

if tenEighty === alsoTenEighty {
    print("两个VideoMode相等")
}

// 两个VideoMode相等

    如上 两个对象都是指向同同一块内存因此是相等的

let tenEighty = VideoMode()
tenEighty.resolution = hd
tenEighty.interlaced = true
tenEighty.name = "1080i"
tenEighty.frameRate = 25.0

let alsoTenEighty = VideoMode()
tenEighty.resolution = hd
tenEighty.interlaced = true
tenEighty.name = "1080i"
tenEighty.frameRate = 25.0

if tenEighty === alsoTenEighty {
    print("两个VideoMode相等")
} else {
    print("两个VideoMode不相等")
}
// 两个VideoMode不相等

虽然两个对象值都是一样,但是初始化在不同内存因此两个对象不相等

Swift常量变量小记

发表于 2024-01-01

var 变量

  • 声明一个变量
    var value = 0
  • 变量的值可以改变
    value = 1

    变量只能改变同一种类型的值,不能修改为其他类型的值,如下如果value改为字符串就会报错

    value = "字符" // 报错

let 常量

  • 声明一个常量
    let constValue = 10

    常量一旦赋值就不能改变,否则会改变

    constValue = 2 // 报错

常量变量的几种声明方式

  • 默认值
    var value = 10
let constValue = 20
  • 也可以先定义类型然后在赋值
    var x:Int,y:Double,z:String
let eror:String,constValue:Int
x = 10
y = 2.0
z = "z"
eror = "error"
constValue = 10

e 可以表示10几次方

var value = 1.28175e2
print(value) // 128.175

类型转换

var three = 3
var threeDouble = Double(three)
print(threeDouble) // 3.0
var PI = 3.1415
var p:Int = Int(PI)
print(p) // 3

类型别名

typealias Age = Int
var age:Age = 10
print(age) // 10
